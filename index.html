Multi-Agent System

snippet 
from langchain.agents import Tool, AgentExecutor
from langchain.chat_models import ChatOpenAI
from autogen import AssistantAgent, UserProxyAgent

# Define specialized agents
researcher = AssistantAgent(
  name="researcher",
  llm_config={"model": "gpt-4", "temperature": 0},
  system_message="You search and extract findings."
)

analyst = AssistantAgent(
  name="analyst",
  llm_config={"model": "gpt-4", "temperature": 0},
  system_message="You synthesize insights."
)

# Orchestrate multi-agent workflow
coordinator = UserProxyAgent(name="coord")

def run(query: str) -> str:
  coordinator.initiate_chat(researcher, msg=query)
  findings = researcher.last_message()
  return analyst.reply(findings)
